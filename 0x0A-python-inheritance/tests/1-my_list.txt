import unittest
from mylist import MyList  # Assuming your MyList class is defined in a separate "mylist.py" file

class TestMyList(unittest.TestCase):

    def test_print_sorted(self):
        my_list = MyList([3, 1, 2, 5, 4])
        expected_result = [1, 2, 3, 4, 5]
        
        # Redirect stdout to capture the printed output
        import sys
        from io import StringIO
        original_stdout = sys.stdout
        sys.stdout = StringIO()

        my_list.print_sorted()

        # Get the printed output
        printed_output = sys.stdout.getvalue().strip()

        # Reset stdout
        sys.stdout = original_stdout

        # Convert the printed output to a list of integers
        printed_list = [int(x) for x in printed_output.split(',')]

        # Assert that the printed list matches the expected result
        self.assertEqual(printed_list, expected_result)

if __name__ == '__main__':
    unittest.main()
